it ('優惠券存在的話，要廣播 applied。', () => {
    let wrapper = mount(CouponCode)
    
    // 設定假的資料
    wrapper.setData({
        coupons: ['50OFF']
    })
    
    // 手動觸發驗證
    wrapper.vm.validate('50OFF')
    
    // 這裡就是關鍵
    expect(wrapper.emitted().applied).toBeTruthy()
})

再写

export default {
    // ... Vue 的其他設定
    
    methods: {
      validate (code) {
        this.valid = this.coupons.includes(code)
        
        // 廣播 (broadcast)
        if( this.valid )
            this.$emit('applied');
      }
    }
}
